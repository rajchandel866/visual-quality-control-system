# Add this method in your BottleDefectDetector class def detect_defect_batch(self, frame): gray_frame = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY) edges = cv2.Canny(gray_frame, 50, 150) result = cv2.matchTemplate(edges, self.template, cv2.TM_CCOEFF_NORMED) min_val, max_val, min_loc, max_loc = cv2.minMaxLoc(result) threshold = 0.6 match = max_val >= threshold color = (0, 255, 0) if match else (0, 0, 255) h, w = self.template.shape top_left = max_loc bottom_right = (top_left[0] + w, top_left[1] + h) cv2.rectangle(frame, top_left, bottom_right, color, 2) cv2.putText(frame, "OK" if match else "Defect", (10, 30), cv2.FONT_HERSHEY_SIMPLEX, 1, color, 2) return frame, match, top_left, bottom_right
